    async def get_by_tournament_with_position_crud(
        self, tournament_id: int
    ) -> list[dict]:
        """获取赛事参赛者（带抢位赛位置信息）"""
        from sqlalchemy import text

        sql = text("""
            SELECT
                p.id,
                p.student_id,
                s.name as student_name,
                p.current_position,
                p.seed_rank
            FROM badminton_tournament_participant p
            JOIN badminton_student s ON p.student_id = s.id
            WHERE p.tournament_id = :tournament_id
            AND p.is_withdrawn = false
            ORDER BY p.current_position NULLS LAST, p.id
        """)

        result = await self.auth.db.execute(sql, {"tournament_id": tournament_id})
        rows = result.mappings().all()
        return [dict(row) for row in rows]

    async def init_positions_crud(
        self, tournament_id: int
    ) -> list[dict]:
        """随机抽签初始化位置"""
        from sqlalchemy import text
        import random

        # 获取所有参赛者
        sql = text("""
            SELECT id FROM badminton_tournament_participant
            WHERE tournament_id = :tournament_id
            AND is_withdrawn = false
        """)
        result = await self.auth.db.execute(sql, {"tournament_id": tournament_id})
        participant_ids = [row[0] for row in result.fetchall()]

        if len(participant_ids) < 2:
            raise CustomException(msg="参赛人数不足，无法初始化位置")

        # 随机打乱
        random.shuffle(participant_ids)

        # 批量更新位置
        updates = []
        for i, pid in enumerate(participant_ids):
            updates.append(f"({pid}, {i + 1})")

        update_sql = text(f"""
            UPDATE badminton_tournament_participant AS p
            SET current_position = v.pos
            FROM (VALUES {', '.join(updates)}) AS v(id, pos)
            WHERE p.id = v.id
        """)
        await self.auth.db.execute(update_sql)
        await self.auth.db.flush()

        return await self.get_by_tournament_with_position_crud(tournament_id)

    async def swap_positions_crud(
        self, participant_id_1: int, participant_id_2: int
    ) -> None:
        """交换两个参赛者的位置"""
        from sqlalchemy import text

        # 获取当前位置
        sql = text("""
            SELECT id, current_position FROM badminton_tournament_participant
            WHERE id IN (:id1, :id2)
        """)
        result = await self.auth.db.execute(sql, {"id1": participant_id_1, "id2": participant_id_2})
        rows = result.fetchall()

        if len(rows) != 2:
            raise CustomException(msg="参赛者不存在")

        positions = {row[0]: row[1] for row in rows}
        pos1 = positions[participant_id_1]
        pos2 = positions[participant_id_2]

        # 交换
        swap_sql = text("""
            UPDATE badminton_tournament_participant
            SET current_position = CASE
                WHEN id = :id1 THEN :pos2
                WHEN id = :id2 THEN :pos1
            END
            WHERE id IN (:id1, :id2)
        """)
        await self.auth.db.execute(
            swap_sql,
            {"id1": participant_id_1, "id2": participant_id_2, "pos1": pos1, "pos2": pos2},
        )
        await self.auth.db.flush()
```

    @classmethod
    async def create_challenge_service(
        cls,
        tournament_id: int,
        challenger_id: int,
        defender_id: int,
        auth: AuthSchema,
    ) -> dict:
        """创建挑战"""
        from sqlalchemy import text, select
        from app.plugin.module_badminton.tournament.model import TournamentParticipantModel

        # 校验赛事类型
        result = await auth.db.execute(
            text("SELECT tournament_type FROM badminton_tournament WHERE id = :tid"),
            {"tid": tournament_id},
        )
        row = result.fetchone()
        if not row or row[0] != "PROMOTION_RELEGATION":
            raise CustomException(msg="仅定区升降赛支持此操作")

        # 获取两个参赛者信息
        stmt = select(
            TournamentParticipantModel.id,
            TournamentParticipantModel.tournament_id,
            TournamentParticipantModel.current_position,
            TournamentParticipantModel.is_withdrawn,
        ).where(TournamentParticipantModel.id.in_([challenger_id, defender_id]))
        result = await auth.db.execute(stmt)
        participants = result.fetchall()

        if len(participants) != 2:
            raise CustomException(msg="参赛者不存在")

        participant_map = {p.id: p for p in participants}
        challenger = participant_map[challenger_id]
        defender = participant_map[defender_id]

        # 校验是否属于同一赛事
        if challenger.tournament_id != tournament_id or defender.tournament_id != tournament_id:
            raise CustomException(msg="参赛者不属于该赛事")

        # 校验是否退赛
        if challenger.is_withdrawn or defender.is_withdrawn:
            raise CustomException(msg="选手已退赛，无法发起挑战")

        # 校验位置：挑战者必须是 defender 的相邻低位
        if challenger.current_position is None or defender.current_position is None:
            raise CustomException(msg="选手尚未分配位置，请先初始化")

        if challenger.current_position != defender.current_position + 1:
            raise CustomException(msg="只能挑战相邻高位的选手")

        # 检查是否有未完成的比赛
        existing_sql = text("""
            SELECT COUNT(*) FROM badminton_tournament_match
            WHERE tournament_id = :tid
            AND round_type = 'PROMOTION_RELEGATION'
            AND status IN ('SCHEDULED', 'IN_PROGRESS')
            AND (
                (player1_id = :c1 AND player2_id = :d1)
                OR (player1_id = :d1 AND player2_id = :c1)
            )
        """)
        result = await auth.db.execute(
            existing_sql,
            {"tid": tournament_id, "c1": challenger_id, "d1": defender_id},
        )
        existing_count = result.scalar()
        if existing_count and existing_count > 0:
            raise CustomException(msg="两选手之间有未完成的比赛")

        # 创建比赛
        match_crud = TournamentMatchCRUD(auth)
        match = await match_crud.create_match_crud(
            tournament_id=tournament_id,
            round_type="PROMOTION_RELEGATION",
            round_number=1,
            match_number=1,  # 抢位赛不区分轮次和场次编号，统一用1
            player1_id=challenger_id,
            player2_id=defender_id,
        )

        return {
            "id": match.id,
            "player1_id": match.player1_id,
            "player2_id": match.player2_id,
            "round_type": "PROMOTION_RELEGATION",
            "status": match.status.value if hasattr(match.status, "value") else match.status,
        }

                "id": row["id"],

    @classmethod
    async def _handle_pr_position_swap(
        cls, match_id: int, winner_id: int | None, auth: AuthSchema
    ) -> None:
        """定区升降赛：挑战者获胜则交换位置"""
        from sqlalchemy import select
        from app.plugin.module_badminton.tournament.model import TournamentMatchModel

        if not winner_id:
            return

        # 获取比赛信息
        result = await auth.db.execute(
            select(
                TournamentMatchModel.id,
                TournamentMatchModel.round_type,
                TournamentMatchModel.player1_id,
                TournamentMatchModel.player2_id,
            ).where(TournamentMatchModel.id == match_id)
        )
        match = result.first()
        if not match:
            return

        round_type = match.round_type
        if hasattr(round_type, "value"):
            round_type = round_type.value

        if round_type != "PROMOTION_RELEGATION":
            return

        # player1 是挑战者
        challenger_id = match.player1_id
        defender_id = match.player2_id

        # 如果挑战者获胜，交换位置
        if winner_id == challenger_id:
            logger.info(
                f"[_handle_pr_position_swap] 挑战者获胜，交换位置: challenger={challenger_id}, defender={defender_id}"
            )
            await TournamentParticipantCRUD(auth).swap_positions_crud(
                challenger_id, defender_id
            )
        else:
            logger.info(
                f"[_handle_pr_position_swap] 守位者获胜，位置不变: defender={defender_id}"
            )
```

// 加载位置板
async function loadPositions() {
  if (!currentTournament.value) return;
  try {
    const res = await TournamentAPIExtended.getPositions(currentTournament.value.id);
    positionBoardData.value = res.data?.data || [];
  } catch (error) {

  }
}

// 加载抢位赛比赛记录
async function loadPRMatches() {
  if (!currentTournament.value) return;
  try {
    const res = await TournamentAPIExtended.getPRMatches(currentTournament.value.id);
    prMatches.value = res.data?.data || [];
  } catch (error) {
    console.error("加载抢位赛比赛记录失败:", error);
  }

// 随机抽签入位
async function handleInitPositions() {
  if (!currentTournament.value) return;
  try {
    await TournamentAPIExtended.initPositions(currentTournament.value.id);
    ElMessage.success("位置初始化成功");
    await loadPositions();
    await loadPRMatches();
  } catch (error: any) {
    console.error(error);
    ElMessage.error(error?.response?.data?.msg || "位置初始化失败");
  }
}

// 发起挑战
async function handleCreateChallenge(challengerId: number, defenderId: number) {
  if (!currentTournament.value) return;
  try {
    await TournamentAPIExtended.createChallenge(
      currentTournament.value.id,
      challengerId,
      defenderId
    );
    ElMessage.success("挑战创建成功");
    await loadPRMatches();
  } catch (error: any) {
    console.error(error);
    ElMessage.error(error?.response?.data?.msg || "创建挑战失败");
  }
}

// 录入抢位赛比分
function handleRecordPRScore(match: any) {
  selectedMatch.value = {
    id: match.id,
    player1: { id: match.player1_id, name: match.player1_name },
    player2: { id: match.player2_id, name: match.player2_name },
    scores: match.scores?.sets || [],
    status: match.winner_id ? "COMPLETED" : "SCHEDULED",
  } as any;
  scoreDialogVisible.value = true;
}

// 格式化抢位赛比分
function formatPRScore(scores: any): string {
  if (!scores || !scores.sets || scores.sets.length === 0) {
    return "-";
  }
  return scores.sets
    .map((s: any) => `${s.player1 || 0}:${s.player2 || 0}`)
    .join(", ");
}
